create table QUESTIONS
(
  ID                        NUMBER(10) not null,
  LAST_MODIFIED_USER_ID     NUMBER(10) not null,
  CREATOR_USER_ID           NUMBER(10) not null,
  MEDIA_ID                  NUMBER(10) not null,
  QUESTION_SET_ID           NUMBER(10) not null,
  QUESTION_TYPE_ID          NUMBER(10) not null,
  TITLE                     VARCHAR2(500),
  DISPLAY_ORDER             NUMBER(5),
  OUTPUT_TITLE              VARCHAR2(500),
  VISIBLE                   NUMBER(1) default 1 not null,
  PUBLISHED                 NUMBER(1) default 1 not null,
  CREATION_DATE             TIMESTAMP default CURRENT_TIMESTAMP,
  LAST_MODIFIED_DATE        TIMESTAMP default CURRENT_TIMESTAMP,
  EXAMPLE_CONTENT           VARCHAR2(4000),
  SHOW_EXAMPLE_AFTER_SUBMIT NUMBER(1) default 1 not null,
  SAVE_HISTORY              NUMBER(1) default 1 not null,
  SHOW_HISTORY              NUMBER(1) default 1 not null,
  SHOW_RESPONSE_CLEAR       NUMBER(1) default 1 not null
);
